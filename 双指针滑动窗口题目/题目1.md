//滑动窗口可以用于解决在一定范围内对字符或元素计数的问题
///滑动窗口维护任意字符元素至少为k次
//l 到 r 为滑动窗口的范围
//l和r只会递增，最坏走2 * n 次，时间复杂度变为O(n)
import java.io.*;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter pw = new PrintWriter(System.out);

        StringTokenizer st = new StringTokenizer(br.readLine());
        int n = Integer.parseInt(st.nextToken());
        int k = Integer.parseInt(st.nextToken());
        int res = n + 1;
        String s = br.readLine();
        if(k == 1){
            pw.println(1);
            br.close();
            pw.flush();
            pw.close();return;
        }
        int[]cnt = new int[26];
        int r = 0,l = 0;
        for(;r < n;r++){
            int c = s.charAt(r) - 'a';
            cnt[c]++;
            while (cnt[c] >= k && l < r){//开始维护窗口，（l- r 满足答案性质，可以缩小窗口到边界）
                    res = Math.min(res,r - l + 1);//记录答案
                    cnt[s.charAt(l) - 'a']--;
                    l++;
                    if(cnt[c] < k){
                        break;
                    }

            }
        }
        pw.println(res == n + 1?-1:res) ;
        br.close();
        pw.flush();
        pw.close();
    }
}
