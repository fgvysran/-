class Solution {

        int n = s.length();

            }

    }

    public int[][] reconstructQueue(int[][] people) {
        Arrays.sort(people,(a,b)->{
            if(a[0] != b[0]){
                return a[0] >= b[0]?-1:1;
            }
            return a[1] <= b[1]?-1:1;
        });
        int n = people.length;
        for(int i = 0;i < n;i++){
            int h = people[i][0],k = people[i][1];
